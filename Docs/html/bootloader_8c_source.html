101 void Timerms(unsigned int nb) {
102  /*
103  * Wait for a while just to be sure that all other modules are started.
104  * To do that we use timer 0.
105  */
106  #if MAINCLOCK/100 <= 0xFF
107  TCCR0B |= 0x01;
108  unsigned int ticksCount = 255 - (MAINCLOCK/1000);
109  #elif(MAINCLOCK/8)/1000 <= 0xFF
110  TCCR0B |= 0x02;
111  unsigned int ticksCount = 255 - ((MAINCLOCK/8)/1000);
112  #elif(MAINCLOCK/64)/1000 <= 0xFF
113  TCCR0B |= 0x03;
114  unsigned int ticksCount = 255 - ((MAINCLOCK/64)/1000);
115  #elif(MAINCLOCK/256)/1000 <= 0xFF
116  TCCR0B |= 0x04;
117  unsigned int ticksCount = 255 - ((MAINCLOCK/256)/1000);
118  #elif(MAINCLOCK/1024)/1000 <= 0xFF
119  TCCR0B |= 0x05;
120  unsigned int ticksCount = 255 - ((MAINCLOCK/1024)/1000);
121  #else
122  #error *** Incompatible Clock set at MAINCLOCK. ***
123  #endif
124  /* Wait 10ms */
125  unsigned int i;
126  for (i = 0; i < nb; i++) {
127  TCNT0 = ticksCount;
128  while (TCNT0 >= ticksCount);
129  }
130 }

00100 void Timerms(unsigned int nb) {
00101     /*
00102      * Wait for a while just to be sure that all other modules are started.
00103      * To do that we use timer 0.
00104      */
00105     #if MAINCLOCK/100 <= 0xFF
00106         TCCR0B |= 0x01;
00107         unsigned int ticksCount = 255 - (MAINCLOCK/1000);
00108     #elif(MAINCLOCK/8)/1000 <= 0xFF
00109         TCCR0B |= 0x02;
00110         unsigned int ticksCount = 255 - ((MAINCLOCK/8)/1000);
00111     #elif(MAINCLOCK/64)/1000 <= 0xFF
00112         TCCR0B |= 0x03;
00113         unsigned int ticksCount = 255 - ((MAINCLOCK/64)/1000);
00114     #elif(MAINCLOCK/256)/1000 <= 0xFF
00115         TCCR0B |= 0x04;
00116         unsigned int ticksCount = 255 - ((MAINCLOCK/256)/1000);
00117     #elif(MAINCLOCK/1024)/1000 <= 0xFF
00118         TCCR0B |= 0x05;
00119         unsigned int ticksCount = 255 - ((MAINCLOCK/1024)/1000);
00120     #else
00121         #error *** Incompatible Clock set at MAINCLOCK. ***
00122     #endif
00123     /* Wait 10ms */
00124     unsigned int i;
00125     for (i = 0; i < nb; i++) {
00126         TCNT0 = ticksCount;
00127         while (TCNT0 >= ticksCount);
00128     }
